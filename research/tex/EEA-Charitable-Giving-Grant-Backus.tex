\frametitle{Empirical Question}
\begin{enumerate}
	
	\item Charity donations tax deductible for tax \textbf{Itemisers} in US
	
	\vspace{1pt}
	\item Price  of donating \$1  function of\textbf{ MTR ($\tau$)} \& \textbf{Itemisation Status} (I) 

		\vspace{10pt}
	 \item[] \[\textrm{Price} = \left\{
	 \begin{array}{lr}
	 1-\tau & : \textrm{Itemiser}\\
	 1 & :\textrm{Non-Itemiser}
	 \end{array}
	 \right.
	 \]
	
	\item[] \colorbox{yellow}{\parbox{\textwidth}{ Question: What is the  elasticity of Donations w.r.t   Price?}}
		\vspace{1pt}
	\footnotesize	
	\item[] \colorbox{green}{Policy Relevance} Charitable Tax Subsidies  \$134bn in US 2010 (Lowry, 2014) \vspace{8pt}

	\item[-]  Is the charity tax subsidy an Optimal Subsidy in the US?
	\item[] e.g Optimal Subsidy is 8 times larger if elasticity is -1 relative to -0.5 Saez (2004)
	
\end{enumerate}
\end{frame}
	
		
		\begin{frame}
			\frametitle{Itemisation}
			\begin{itemize}
			\item $\highlight{E_{it}}$ all other tax-deductible expenses (e.g medical costs ) (\$)
				\vspace{15pt}
				\item  $\highlight{S_{it}}$	tax-free allowance ($\approx$ \$6000 for singles, \$12000  married couples)
				\vspace{15pt}
				\item  $\highlight{\Delta_{it}}=\max\{D_{it}+E_{it}-S_{it},0\}$ is \textbf{tax deductible }
				\vspace{15pt}
				\item  $\Delta_{it}>0$ agent may itemise (tax-file) and tax-deduct $\Delta_{it}$
				\vspace{15pt}
				\item[]	
				$\Delta_{it}>0 \rightarrow  \left\{
				\begin{array}{lr}
				E_{it} > S_{it}& : \highlight{\textrm{Exogenous Itemiser}}\\
				E_{it}\leq S_{it}& : \highlight{ \textrm{Endogenous Itemiser}}
				\end{array}
				\right.$
				\item[] $\Delta_{it}=0 $ \highlight{ \textrm{Non Itemiser}}
			\end{itemize}\end{frame}

	\begin{frame}
		\frametitle{ Bias From `Switchers': Intuition}
		\begin{itemize}
		
			\item[]  \begin{equation*}\Delta \log(D_{it})=\beta \Delta\log(P_{it})+ \omega'\Delta X_{it}+ u_{it} \end{equation*}
		\vspace{1pt}

\item[] \textbf{ Stop Itemiser} $ \Delta I_{it}=-1, I_{i,t-1}=0, \;I_{it}=1$
\vspace{10pt}
\item   $\Delta \log (P_{it})=-\log(1-\tau_{i,t-1})>0$
\vspace{10pt}
\item  $\Delta \log (D_{it})=\log( D_{it})-\log (D_{i,t-1})\leq \log(S_{it}+E_{it})$
\vspace{5pt}
\item[]
\vspace{5pt}
\item[] Since $I_{it}=0$ then  $D_{it}\leq S_{it}+E_{it}$ by definition and $-\log (D_{i,t-1})\leq 0$.
\vspace{5pt}
\begin{center}
\item[] \textbf{ $u_{it}$  bounded above \&  -ve correlation with $\Delta \log P_{it}$}
\end{center}


	



\end{itemize}
\end{frame}


\begin{frame}
	\frametitle{Theorem 1: Downard Bias From Non-Itemisers/Switchers}

	\begin{itemize}
	\centering
	\item[] 	$p_{1}=$$\mathcal{P}$$\{\Delta I_{it}=1\}$ $\;\;\;\;\;\;\;\;\;\;\;p_{-1}=$$\mathcal{P}$$\{\Delta I_{it}=-1\}$ \vspace{10pt}
	\item[] $\xi_1=\frac{Cov(u_{it},\log(P_{it})|\Delta I_{it}=1))}{Var(\log(P_{it}))}$ $\;\;\;\;\;\;\;\;\xi_{-1}=\frac{Cov(u_{it},\log(P_{it})|\Delta I_{it}=-1))}{Var(\log(P_{it}))}$

	\vspace{25pt}
\item[]	\framebox{\textsc{Theorem\,1:\,\,} $\hat{\beta}_{FD}\overset{p}{\rightarrow}\beta+\underset{<0}{\underbrace{p_{1}\xi_1+p_{-1}\xi_{-1}}}$	where $\xi_1, \xi_{-1}<0$}
		
\vspace{25pt}
\item $p_1$ (probability of Start Itemiser $\approx 0.10$)	
\item $p_{-1}$	 (probability of Stop Itemiser $\approx 0.08$)
\item[]  $\approx 18\%$ of sample are Switchers





\end{itemize}

\end{frame}

\begin{frame}
	\frametitle{Theorem 2: Consistent inference Controlling for $\Delta I_{it}$ }
	\begin{itemize}
\small
\item[] \highlight{\textrm{Itemiser Model:}} $\;\;\Delta \log(D_{it})=\gamma\Delta I_{it}+\beta \Delta\log(P_{it})+ \omega'\Delta X_{it}+ \eta_{it} $

\begin{equation*}\scriptsize
\hat{\theta}_{FD}^{I}=\left(\sum_{i=1}^{N}\sum_{t=2}^{T}w_{it}w_{it}'\right)^{-1}\sum_{i=1}^{N}\sum_{t=2}^{T}w_{it}\Delta\log(D_{it})
\end{equation*}
 $w_{it}=(\Delta I_{it}, \Delta \log (P_{it}), \Delta X_{it}')'$ 
 $\;\;\;\;\hat{\theta}^I_{FD}=(\hat{\gamma}_{FD}^I,\hat{\beta}_{FD}^I,\hat{\omega}_{FD}^{I'})'$.
\vspace{15pt}
\item[]  $\bar{\tau}_{1}=E[\Delta\log(P_{it})|\Delta I_{it}=1]$\\ \vspace{2pt}
 $\bar{\tau}_{-1}=E[\Delta \log(P_{it})|\Delta I_{it}=-1]$\\  \vspace{2pt}
$C=\det(E[w_{it}w_{it}'])$ 
\vspace{10pt}
\item[] \textsc{Theorem\,2\,\,}\framebox{ $
\hat{\beta}_{FD}^{I}\overset{p}{\rightarrow}\beta+\frac{p_{1}p_{-1}}{C}(\bar{\tau}_{1}+\bar{\tau}_{-1})(E[\eta_{it}|\Delta I_{it}=-1]+E[\eta_{it}|\Delta I_{it}=1]) $}

\end{itemize}
\end{frame}

\begin{frame}
	\frametitle{MTR and price construction}
	\begin{itemize}
		
		

\item[] \textit{Actual Price:} \framebox{$P_{it}^a=1-I_{it}\tau_{it}$}
\item[] \textit{Auten et al (2002) Price}: \framebox{$P_{it}^b=1-\tau_{it}^{b}$ } \vspace{10pt}

---------------------------------------------------------------------------------
\vspace{10pt}
\item[]$\tau_{it}^{s}$: \textbf{'Synthetic' MTR}  using $i$'s tax  characteristics in $t$, including giving
set to 0, but the tax code from  $t-2$. \vspace{9pt}

\item[] $\tau_{it}^{f}$: MTR applied to first dollar of giving \vspace{9pt}
\item[]  $\tau_{it}^{p}$:  MTR for giving set at 1\% household income \vspace{9pt}
\item[]  $\tau_{it}^{b}$: \textbf{'Auten MTR'} $\tau_{it}^{b}=\frac{1}{2}(\tau_{it}^{f}+\tau_{it}^{p})$


\end{itemize}\end{frame}
