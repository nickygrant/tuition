		\begin{frame}
			
			\frametitle{Time Series Process vs. Realisation}
			
			\begin{itemize}
				\normalsize		
				\item[]\textbf{Time Series Process: } \{$Y_t$: $t\in \mathcal{T}\}$ for some set $\mathcal{T}$
				\pause
				\vspace{10pt}
				\item[] \hspace{25pt} Unless stated otherwise assume $\mathcal{T}=\{\cdots -2,-1,0,1,2,\cdots\}$
				\item[] \hspace{25pt} Let $\{Y_t\}$ denote $\{Y_t: t\in \mathcal{T}\}$
				\pause
				\vspace{25pt}
				\item[]   $\{Y_t\}$ is multivariate r.v with a joint distribution fn.	\pause
				\item[] \hspace{25pt} e.g. $Y_t$ is  independently distributed $N(0,1)$ for all $t\in \mathcal{T}$
				
				\pause
				\vspace{25pt}
				\item[] \textbf{Realisation:} $\{y_t^s: t\in \mathcal{T}\}$ in \textit{state} $s$ 	 drawn from  $\{Y_t\}$ 
				\vspace{25pt}
				\pause
				\item[] \textbf{Sample Data:} $(y_1,\cdots,y_T)'$  realisation of sample size T from  $\{Y_t\}$ 
				\pause
				
				
			\end{itemize}
		\end{frame}

		\section{Fundamental Statistical Definitions}
		
		\begin{frame}
			\frametitle{Distribution of a Time Series Process}
			\begin{itemize}
				
				\item[] \normalsize  \hspace{-5pt}\textbf{ \large Process generating sample data} $\large \{Y_{t} : t \in \mathcal{T}\} $ for $\large\mathcal{T}=\{1,\hdots, T\}$	
				\pause
				\vspace{20pt}
				\item[]\hspace{15pt}{\makebox[4cm]{\large$f_{\mathcal{T}}(y_1,\cdots,y_T)$\hfill} \textbf{Joint density function} of $\{Y_{t} : t \in \mathcal{T}\}$
				}
				
				\pause
				\vspace{20pt}
				
				\item[]\hspace{15pt}{\makebox[4cm]{\large$f_{t}(y_t)$\hfill} \textbf{Marginal density function} of $Y_t$ (for  $t \in \mathcal{T}$)  }
				
				\pause
				\vspace{30pt}
				\large
				\item[]  $(y_1^s,\cdots,y_T^s)'$ in state \textit{s} drawn from  joint distn. of $\{Y_{t} : t \in \mathcal{T}\}$
				\pause
				\vspace{12pt}
				\item[] $y_t^s$ in state s drawn from marginal distn. of $Y_t$
				\vspace{18pt}
				\pause
				\item[] We observe \textbf{one realisation}, the sample data  $(y_1,\cdots, y_T)'$
			\end{itemize}
			
			
			
		\end{frame}
		
		
		
		
		\begin{frame}
			\frametitle{Moments of Time Series Process (univariate)}
			\begin{itemize}
				
				\large
				
				\item[]\begin{myexx*}[width=5.4in]{Population Moments of $Y_t$} For any function $g(\cdot)$
					\begin{equation*}\mathbb{E}[g(Y_t)] := \int_{-\infty}^{\infty} g(y) f_t(y)dy  \end{equation*}
				\end{myexx*}
				\pause
				\item[] The \textbf{average in the population of} $g(Y_t)$
				\vspace{12pt}\pause
				
				\item[]\hspace{12pt} {\makebox[3cm]{\textbf{Mean} of $Y_t$\hfill}} {\makebox[5cm]{$\mathbb{E}[Y_t]$\hfill}} \hspace{-8pt} {\makebox[6cm]{$\mu_t:= \int_{-\infty}^{\infty} y f_t(y)dy $\hfill}} 
				\vspace{18pt}\pause
				\item[]\hspace{12pt} {\makebox[3cm]{\textbf{Var} of $Y_t$\hfill}} {\makebox[5cm]{$\mathbb{E}[(Y_t-\mu_t)^2]$\hfill}} \hspace{-8pt} {\makebox[6cm]{Var($Y_t$):$= \int_{-\infty}^{\infty} (y-\mu_t)^2 f_t(y)dy $}} 
			
			\end{itemize}
		\end{frame}
		
		\begin{frame}
			\frametitle{Moments of Time Series Process (bivariate)}
			\begin{itemize}
				
				\large
				\item[] Consider $\mathcal{T}=\{t_1,t_2\}$ where $t_1\neq t_2 \in \mathcal{T}$ 	\vspace{8pt}
				
				\item[]\begin{myexx*}[width=5.4in]{Population Moments of $(Y_{t_1},Y_{t_2})$}
					\begin{equation*}\mathbb{E}[g(Y_{t_1},Y_{t_2})] := \int_{-\infty}^{\infty}\int_{-\infty}^{\infty} g(y_{t_1},y_{t_2}) f_{\mathcal{T}}(y_{t_1},y_{t_2})dy_{t_1}dy_{t_2}  \end{equation*}
				\end{myexx*}
				\pause
				\item[] The \textbf{average in the population of} $g(Y_{t_1}, Y_{t_2})$
					\vspace{10pt}
			
				
			
				\vspace{15pt}
				
				\pause
				\item[]\hspace{29pt} Cov$(Y_{t_1},Y_{t_2}):=\mathbb{E}[(Y_{t_1}-\mu_{t_1})(Y_{t_2}-\mu_{t_2})]$ \hspace{10pt} \textbf{Covariance}  $Y_{t_1},Y_{t_2}$
				\vspace{10pt}\pause
				\item[]$\hspace{81.2pt} = \int_{-\infty}^{\infty}\int_{-\infty}^{\infty} (y_{t_1}-\mu_{t_1})(y_{t_2}-\mu_{t_2}) f_{\mathcal{T}}(y_{t_1},y_{t_2})dy_{t_1}dy_{t_2} $
				
				
			\end{itemize}
		\end{frame}
		
		
		
		
		
		
		
		
		\begin{frame}
			
			\frametitle{Stationarity}
			\large
			\begin{itemize}
				\item[]\begin{mydef*}[width=5.5in]{Weakly Stationary Process}  A process $\{Z_t\}$   is \textbf{weakly stationary} if
					\vspace{4pt}
					\begin{enumerate}\large
							
						\item[] \makebox[6.23cm]{ $\mathbb{E}[Z_t] =\mu $ \hfill} for all $t\in\mathcal{T}$,\medskip
						
						\item[] \makebox[6.23cm]{ Var($Z_t$)$ =\sigma^2<\infty$\hfill}  for all $t\in\mathcal{T}$,
						\medskip
						
						\item[] \makebox[6.23cm]{ Cov($Z_{t_1}, Z_{t_2}$) $=\gamma(|t_1-t_2|)$\hfill} for all  $t_1,t_2\in\mathcal{T}$.
					\end{enumerate}
				\end{mydef*}
				
				\pause
				\vspace{25pt}
				
					\item[]\begin{mydef*}[width=5.5in]{Strictly Stationary Process}  A process $\{Z_t\}$   is \textbf{strictly stationary} if for all $t_1,\cdots t_k \in \mathcal{T}$ (for any positive integer k) the joint density of $(Z_{t_1+\tau},...,Z_{t_k+\tau})'$ is \textbf{invariant} to $\tau$.
						
			
					\end{mydef*}	
				
				
				\end{itemize}
			\end{frame}
		
				
				\begin{frame}
					
					\frametitle{Stationarity}
					\normalsize
				\begin{itemize}
				\item[] \textbf{Strict stationarity} is a stronger condition than weak stationarity
				\vspace{18pt}\pause
				\item[] We refer to weak stationarity as stationarity (unless stated otherwise)
				\vspace{18pt}\pause
				
				\item[]\textbf{Process generating sample data may not be stationary, e.g.}
				\pause\vspace{5pt}
				\begin{enumerate}\normalsize
					\item[]\hspace{10pt} Mean	 and/or (co-) variance of $Z_t$ is not constant over time
					\pause\hspace{10pt} 
					\item[]\hspace{10pt}  $Z_t$ has a unit root (e.g. a random walk)
					
				\end{enumerate}
				\end{itemize}

			
			
		\end{frame}
		
		\begin{frame}
			\frametitle{White Noise Process}
			\large
			
			\begin{itemize}
				\item[]\large \begin{mydef*}[width=5in]{White Noise} A process $\{Z_t\}$   is \textbf{white noise} if
					\medskip
					\begin{enumerate}\large
						\item[] \makebox[4cm]{ $\mathbb{E}[Z_{t}]=0$ \hfill} for all $t\in\mathcal{T}, \medskip $
						
						
						\item[] \makebox[4cm]{\noindent $\mathbb{E}[Z_{t}^{2}]=\sigma ^{2}<\infty$} for all $t\in\mathcal{T}, $\medskip
						
						\item[] \makebox[4cm]{ $\mathbb{E}[Z_{t_1}Z_{t_2}]=0$ \hfill} for all $t_1\neq t_2 \in \mathcal{T}$.\medskip
					\end{enumerate}
				\end{mydef*}
				
				\item[] i.e. \textbf{zero mean, constant variance and no correlation across time}
				\pause
				
				\vspace{10pt}
				\item[] WN($\sigma^2$) \hspace{10pt} notation for a \textit{white noise process with variance} $\sigma^2$
				\vspace{10pt}
				\pause
				\item[] \textbf{Note:} white noise $\Rrightarrow$  weak stationarity (not  the reverse)
			\end{itemize}
			
		\end{frame}
		
\begin{frame}
	\frametitle{Auto-covariance \& Autocorrelation Function}
	\normalsize
	
	\begin{itemize}
		\item[]  $\{Y_t\}$ assumed stationary (unless stated otherwise)
		\pause \vspace{10pt}
		\item[]	\begin{mydef*}[width=4.9in]{Autocovariance Function of $\{Y_t\}$}
			\makebox[7cm]{$\gamma(k):=$Cov$(Y_t,Y_{t-k})$\hfill} $\forall \; k \in \mathcal{T}$
		\end{mydef*}
		\vspace{8pt}\pause
		\item[]	\begin{mydef*}[width=4.9in]{Autocorrelation Function of $\{Y_t\}$}
			\makebox[7cm]{$\rho(k):=\frac{\gamma(k)}{\gamma(0)}$\hfill} $\forall \;  k \in \mathcal{T}$
		\end{mydef*}
		\pause
		\vspace{14pt}
		\item[] \hspace{15pt} $\rho(k)$ \textbf{correlation} between $Y_t$ and $Y_{t-k}$ 
		\pause
		\vspace{8pt}
		\item[]\hspace{15pt} Correlations a function of 'gap' k, not time (stationarity)
		
		\pause 	\vspace{5pt}\pause
		\item[]  \hspace{15pt}  Correlations in sample  data should reflect the correlation in $\{Y_t\}$ (for large T)

	\end{itemize}
\end{frame}

		\frametitle{Importance of MA($\infty)$: Wold Decomposition}
		
		\begin{itemize}
			\item[] \begin{mytheo*}{Wold Decomposition}
				Any  stationary process can be
				represented as $MA(\infty ):$
				\begin{equation*}
				Y_{t}=\alpha +\sum_{s=0}^{\infty }\theta _{s}\varepsilon _{t-s},\quad \theta
				_{0}=1 \qquad \varepsilon_t \sim WN(\sigma^2)
				\end{equation*}%
				for some coefficients $\theta_1,..,\theta_{\infty}$ satisfying \textbf{square summability} \end{mytheo*}
			
			\pause
			
			\vspace{20pt}
			\item[]  Can derive mean, variance and covariances of a stationary process by finding its MA($\infty$) form and using general formulas for mean, variance and covariance of an MA($\infty$)
			
			
			\vspace{20pt}
			\item[] We use this method for the stationary AR(1) process (see \textbf{Lecture Notes 1})
		\end{itemize}

			\end{itemize}\begin{center}

			\end{center}
			\pause
	
			
			\vspace{-0.5cm}
			
			\begin{minipage}{.5\linewidth}
				\begin{eqnarray*}
				{\color{red}	(1-L)(1+L)Y_t} &=&(1-L)(Y_t+LY_t)\\ \pause
					&=&(1-L)(Y_t+Y_{t-1})\\\pause
					&=&(Y_t+Y_{t-1})-L(Y_t+Y_{t-1})\\\pause
					&=& (Y_t+Y_{t-1})-Y_{t-1} -Y_{t-2}\\\pause
					&=&Y_{t}-Y_{t-2}
				\end{eqnarray*}
			\end{minipage}\pause

						{\color{red}(1-L^2)Y_t} &=& Y_t-L^2 Y_{t}\\\pause
